    <code value="PST"/>
    <display value="Postal Addres"/>
    <designation>
        <language value="nl-NL"/>
        <use>
            <system value="http://snomed.info/sct"/>
            <code value="900000000000013009"/>
            <display value="Synonym"/>
        </use>
        <value value="Postadres"/>
    </designation>
</concept>
