If it is every 2 minutes, this would work:

Melva Peters (Jun 19 2018 at 14:34):

<dosageInstruction>
<text value="one to two tablets every two minutes"/>
<timing>
<repeat>
<frequency value="1"/>
<period value="2"/>
<periodUnit value="min"/>
</repeat>
</timing>
<doseAndRate>
<type>
<coding>
<system value="http://hl7.org/fhir/dose-rate-type"/>
<code value="ordered"/>
<display value="Ordered"/>
</coding>
</type>
<doseRange>
<low>
<value value="1"/>
<unit value="TAB"/>
<system value="http://hl7.org/fhir/v3/orderableDrugForm"/>
<code value="TAB"/>
</low>
<high>
<value value="4"/>
<unit value="TAB"/>
<system value="http://hl7.org/fhir/v3/orderableDrugForm"/>
<code value="TAB"/>
</high>
</doseRange>
</doseAndRate>
</dosageInstruction>
